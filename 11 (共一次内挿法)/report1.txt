void processing(){

    // 最近隣内挿法の計算
    int d[2][2] = {
        {10, 30},
        {20, 40}
    };

    // 内挿点
    double insertion[2], min_ins[2];

    // 内挿点の値
    int result;

    printf("<使用する4つの画素値>\n");
    for(int i = 0; i < 2; i++){
        for(int j = 0; j < 2; j++){
            printf("%d ", d[j][i]);
        }
        putchar('\n');
    }
    
    printf("内挿点の座標を入力して下さい\n");
    for(int i = 0; i < 2; i++){

        switch (i){
            case 0:{
                printf("水平(0.0以上1.0未満):");
                break;
            }
            case 1:{
                printf("垂直(0.0以上1.0未満):");
                break;
            }
        }

        // 内挿点入力
        scanf("%lf", &insertion[i]);

        // 不正値検出
        if(insertion[i] < 0 || insertion[i] >= 1 ){
            i--;
            continue;
        }
    }

    // 内挿点の座標を算出
    for(int i = 0; i < 2; i++){
        // 1 - 内挿点座標
        min_ins[i] = 1 - insertion[i];
    }

    // 内挿点の値を計算(x)
    double tmp[2];
    for (int i = 0; i < 2; i++){
        // 横
        tmp[i] = (d[0][i] * min_ins[0]) + (d[1][i] * insertion[0]);
    }

    result = (int)((tmp[0] * min_ins[1]) + (tmp[1] * insertion[1]) + 0.5);

    // 内挿点計算
    printf("<内挿点の画素値>\n");
    printf("%d\n", result);

}
