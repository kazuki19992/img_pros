void processing(){
    // int i, j, k, cpmode[3];
    // imgin 2 imgout
    int block;
    int ave = 0;
    int sum = 0;

    printf("ブロックサイズを入力して下さい(2,4,8,16)：");
    scanf("%d",&block);

    for(int i = 0; i < 3; i++){
        // コピーを行う
        for(int j = 0; j < height; j++){
            for(int k = 0; k < width; k++){
               
                if(i == 0){
                    // 4画素ずつに分ける
                    if((j % block == 0) && (k % block == 0)){
                        // 標本化処理
                        for(int row = 0; row < block; row++){
                            for(int columm = 0; columm < block; columm++){
                                sum += imgin[0][j + row][k + columm];
                            }
                        }
                        ave = sum / (block * block);
                        
                        // imgoutへの出力
                        for(int l = 0; l < block; l++){
                            for(int m = 0; m < block; m++){
                                imgout[0][j + l][k + m] = ave;
                            }
                        }
                        sum = 0;
                    }
                    
                }else{
                    // Cb, Crの固定値への置き換えを行うと
                    // モノクロ画像への変換が行える
                    imgout[i][j][k] = 128;
                }
                // k++;
            }
            // j++;
        }

    }
    printf("\n入力画像データをコピーして出力画像データを作成しました.\n");
}
